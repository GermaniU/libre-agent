import hashlib, time
from pathlib import Path

def hash_tree(root: Path) -> dict:
    # SHA-256 de cada archivo bajo root
    return {p: hashlib.sha256(p.read_bytes()).hexdigest()
            for p in root.rglob("*") if p.is_file()}
